Looping over two collections 

names = ['Ann','John','james','lily']
colors =['red','white','xyz']

n = min(len(names),len(colors))

for i in range(n):
  print(names[i],colors[i])

Ann red
John white
james xyz

for i in zip(names,colors):
  print(i)

('Ann', 'red')
('John', 'white')
('james', 'xyz')

for name,color in zip(names,colors):
  print(name,color)

Ann red
John white
james xyz
